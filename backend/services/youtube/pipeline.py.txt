import random
import time
from typing import Dict, List, Optional

from .config import YoutubePipelineConfig, default_channel_ids
from .downloader import YoutubeDownloader
from .fetcher import YoutubeFetcher
from .transcriber import YoutubeTranscriber


class YoutubePipeline:
    def __init__(self, config: Optional[YoutubePipelineConfig] = None):
        self.config = (config or YoutubePipelineConfig()).normalize()
        self.fetcher = YoutubeFetcher(self.config)
        self.downloader = YoutubeDownloader(self.config)
        self.transcriber = YoutubeTranscriber(self.config)

    def collect_video_urls(self, channel_ids: Optional[List[str]], start_date: str) -> List[str]:
        channels = channel_ids or default_channel_ids()
        urls: List[str] = []
        for channel_id in channels:
            urls.extend(self.fetcher.get_videos_after_date(channel_id, start_date))
        return urls

    def process_urls(self, urls: List[str], sleep_min: float = 0.5, sleep_max: float = 1.5) -> List[Dict]:
        results = []
        for url in urls:
            row: Dict = {
                "url": url,
                "downloaded": False,
                "transcribed": False,
                "cached_audio": False,
            }
            print(f"[youtube.pipeline] process_start url={url!r}", flush=True)
            audio = self.downloader.download_audio(url)
            if audio:
                row["downloaded"] = True
                row["audio_path"] = audio
                row["cached_audio"] = bool(getattr(self.downloader, "last_hit_cache", False))
                print(
                    f"[youtube.pipeline] audio_ready url={url!r} audio_path={audio!r} cached_audio={row['cached_audio']!r}",
                    flush=True,
                )
                transcribe_result = self.transcriber.transcribe_audio(audio)
                row["transcribe"] = transcribe_result
                row["transcribed"] = bool(transcribe_result.get("ok"))
                print(
                    f"[youtube.pipeline] transcribe_result url={url!r} ok={transcribe_result.get('ok')!r} "
                    f"txt_path={transcribe_result.get('txt_path')!r} reason={transcribe_result.get('reason')!r}",
                    flush=True,
                )
            else:
                row["transcribe"] = {"ok": False, "reason": "audio_not_found_or_download_failed"}
                print(f"[youtube.pipeline] audio_missing url={url!r}", flush=True)
            results.append(row)
            time.sleep(random.uniform(sleep_min, sleep_max))
        print(f"[youtube.pipeline] process_finished total={len(results)!r}", flush=True)
        return results
