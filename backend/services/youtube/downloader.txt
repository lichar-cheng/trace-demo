import re
import traceback
from pathlib import Path
from typing import Optional
from urllib.parse import parse_qs, urlparse

from .config import YoutubePipelineConfig


def log_downloader(event: str, **kwargs):
    detail = " ".join(f"{key}={value!r}" for key, value in kwargs.items())
    print(f"[youtube.downloader] {event} {detail}".strip(), flush=True)


class YoutubeDownloader:
    def __init__(self, config: YoutubePipelineConfig):
        self.config = config
        self.last_hit_cache = False

    def _extract_video_id(self, url: str) -> str:
        try:
            parsed = urlparse(url)
            if parsed.netloc in {"youtu.be", "www.youtu.be"}:
                return parsed.path.strip("/")
            if "youtube.com" in parsed.netloc:
                if parsed.path == "/watch":
                    return parse_qs(parsed.query).get("v", [""])[0]
                match = re.search(r"/(shorts|embed)/([^/?]+)", parsed.path)
                if match:
                    return match.group(2)
        except Exception as exc:
            log_downloader("extract_video_id_failed", url=url, error=str(exc))
            return ""
        return ""

    def _find_existing_audio(self, video_id: str) -> Optional[str]:
        if not video_id:
            return None

        audio_dir = Path(self.config.audio_output_dir)
        if not audio_dir.exists():
            log_downloader("audio_dir_missing", audio_dir=str(audio_dir))
            return None

        patterns = [
            f"*{video_id}*.mp3",
            f"*{video_id}*.m4a",
            f"*{video_id}*.webm",
            f"*{video_id}*.wav",
            f"*{video_id}*.opus",
        ]
        for pattern in patterns:
            matches = sorted(audio_dir.rglob(pattern), key=lambda item: item.stat().st_mtime, reverse=True)
            if matches:
                log_downloader("cache_hit", video_id=video_id, path=str(matches[0]))
                return str(matches[0])
        log_downloader("cache_miss", video_id=video_id, audio_dir=str(audio_dir))
        return None

    def download_audio(self, url: str) -> Optional[str]:
        self.last_hit_cache = False
        video_id = self._extract_video_id(url)
        log_downloader("download_start", url=url, video_id=video_id, audio_output_dir=self.config.audio_output_dir)

        cached = self._find_existing_audio(video_id)
        if cached:
            self.last_hit_cache = True
            return cached

        try:
            import yt_dlp
        except Exception as exc:
            log_downloader("yt_dlp_import_failed", error=str(exc))
            print(traceback.format_exc(), flush=True)
            return None

        output_dir = Path(self.config.audio_output_dir)
        output_dir.mkdir(parents=True, exist_ok=True)
        ydl_opts = {
            "ffmpeg_location": self.config.ffmpeg_location,
            "format": "bestaudio/best",
            "outtmpl": f"{output_dir}/%(channel)s_%(upload_date)s_%(id)s.%(ext)s",
            "sleep_interval_requests": 1.5,
            "ignoreerrors": False,
            "quiet": False,
            "noprogress": True,
            "no_warnings": False,
            "postprocessors": [{
                "key": "FFmpegExtractAudio",
                "preferredcodec": "m4a",
                "preferredquality": "192",
            }],
        }
        if self.config.proxies:
            ydl_opts["proxy"] = self.config.proxies.get("https") or self.config.proxies.get("http")

        log_downloader("ytdlp_options", ffmpeg_location=self.config.ffmpeg_location, proxy=ydl_opts.get("proxy"), outtmpl=ydl_opts["outtmpl"])

        downloaded_id = video_id
        filename = None
        try:
            with yt_dlp.YoutubeDL(ydl_opts) as ydl:
                info = ydl.extract_info(url, download=True)
                if not info:
                    log_downloader("extract_info_empty", url=url)
                    return None
                downloaded_id = info.get("id") or video_id
                filename = ydl.prepare_filename(info)
                log_downloader("download_finished", url=url, downloaded_id=downloaded_id, prepared_filename=filename)
        except Exception as exc:
            log_downloader("download_exception", url=url, error=str(exc))
            print(traceback.format_exc(), flush=True)
            return None

        if filename:
            file_path = Path(filename)
            converted = file_path.with_suffix(".m4a")
            if converted.exists():
                log_downloader("converted_file_found", path=str(converted))
                self.last_hit_cache = False
                return str(converted)
            if file_path.exists():
                log_downloader("prepared_file_found", path=str(file_path))
                self.last_hit_cache = False
                return str(file_path)

        fallback = self._find_existing_audio(downloaded_id)
        if fallback:
            log_downloader("fallback_found", path=fallback, downloaded_id=downloaded_id)
        else:
            log_downloader("fallback_missing", downloaded_id=downloaded_id, output_dir=str(output_dir))
        self.last_hit_cache = bool(fallback)
        return fallback
